#+begin_src elisp
(provide 'config-tab)

(define-leader! tab "H-t")
(keymap! :tab
         "n" #'tab-new
         "d" #'tab-close
         "t" #'tab-next
         "[" #'tab-previous
         "]" #'tab-next
         "b" #'tab-bar-mode)
#+end_src
