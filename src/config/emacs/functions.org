    ;; Maybe optional dest?
    (defun clone-repo-from-url (repo-url) 
      (let* ((repo-name (car (last (s-split "/" repo-url))))
	     (directory-dest (expand-file-name (concat "$p/" repo-name))))
	     (magit-clone repo-url directory-dest)))
